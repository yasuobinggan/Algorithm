#include <bits/stdc++.h>
using namespace std;
int n;
int a[6];
int pd(){
	int p[20]={0};
	for(int i=0;i<6;i++){
		int k=0;
		while(a[i]){
			p[a[i]%10]++;
			if(p[a[i]%10]>2)return 0;
			a[i]/=10;
			k++;
		}
		if(k<3) return 0;
	}
	return 1;
}

			a[2]=(j/100)*i;a[3]=i*j;

			}
		}
